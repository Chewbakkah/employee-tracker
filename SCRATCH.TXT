SELECT CONCAT(e.first_name, ' ', e.last_name) AS name, r.title FROM employee e JOIN roles r on e.role_id = r.id; 

SELECT e.id, e.first_name, e.last_name, r.title FROM employee e Join roles r on e.role_id = r.id;

SELECT e.id, e.first_name, e.last_name, r.title, r.salary, e.manager_id, d.dept_name FROM employee AS e JOIN roles AS r ON r.id = e.role_id JOIN department AS d ON d.id = r.department_id ORDER BY e.id;
